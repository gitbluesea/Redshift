## 1. Redshift에서 UNLOAD 명령어로 AWS S3에 데이터를 안전하게 내보낸 후, 해당 데이터를 다른 원격 서버로 SFTP(Secure FTP)를 통해 안전하게 전송하는 전체 Shell 스크립트 양식입니다.
대용량 처리를 고려하여, S3에서 스크립트 실행 서버의 디스크(임시 공간)로 다운로드한 뒤 곧바로 SFTP로 전송하고 로컬 파일을 삭제하여 AP 서버의 메모리와 디스크 부하를 최소화하도록 설계했습니다.

### 📋 전체 Shell 스크립트
```
#!/bin/bash
# 에러 발생 시 즉시 스크립트 중단set -e
# ==========================================
# 1. 환경 설정 (사용자 환경에 맞게 수정)
# ==========================================

# Redshift 설정
export PGPASSWORD="your_redshift_password"
REDSHIFT_HOST="://amazonaws.com"
REDSHIFT_PORT="5439"
DB_NAME="your_db"
USER_NAME="your_user"

# S3 및 AWS 설정 (Redshift가 접근할 IAM Role)
S3_BUCKET="s3://your-bucket-name/unload_data/"
IAM_ROLE="arn:aws:iam::123456789012:role/YourRedshiftRole"

# 로컬 임시 디렉토리 (스크립트 실행 서버 내부 디스크)
LOCAL_TEMP_DIR="/tmp/redshift_unload_$(date +%Y%m%d%H%M%S)"
mkdir -p "$LOCAL_TEMP_DIR"

# 원격 SFTP 서버 설정
SFTP_USER="sftp_username"
SFTP_HOST="://remote-server.com"
SFTP_PORT="22"                                # 기본 포트 22, 다르면 수정
SFTP_REMOTE_DIR="/path/to/remote/target/dir"   # 목적지 디렉토리
PRIVATE_KEY_PATH="/home/user/.ssh/id_rsa"     # SFTP 인증용 SSH 개인키 경로

# ==========================================
# 2. Redshift UNLOAD 실행 (S3로 내보내기)
# ==========================================
echo "[1/4] Redshift에서 S3로 UNLOAD 시작..."

# MANIFEST 옵션을 주면 분할 저장된 파일 목록(manifest)이 생성되어 관리가 편합니다.
psql -h "$REDSHIFT_HOST" -p "$REDSHIFT_PORT" -d "$DB_NAME" -U "$USER_NAME" -c "
UNLOAD ('SELECT * FROM large_table WHERE created_at >= \'2026-01-01\'')
TO '$S3_BUCKET'
IAM_ROLE '$IAM_ROLE'
FORMAT AS CSV
GZIP
ALLOWOVERWRITE
MANIFEST;
"

echo "-> UNLOAD 완료."

# ==========================================
# 3. S3에서 로컬 임시 디렉토리로 다운로드
# ==========================================
echo "[2/4] S3에서 AP 서버 로컬 디렉토리로 다운로드 중..."
aws s3 cp "$S3_BUCKET" "$LOCAL_TEMP_DIR" --recursive
echo "-> 다운로드 완료. (위치: $LOCAL_TEMP_DIR)"

# ==========================================
# 4. SFTP를 통해 원격 서버로 전송
# ==========================================
echo "[3/4] 원격 서버($SFTP_HOST)로 SFTP 전송 시작..."

# 패스워드 입력 없이 자동화를 위해 SSH Key인증 방식을 사용합니다.
# BatchMode=yes 옵션은 인증 실패 시 대기하지 않고 즉시 에러를 뱉게 합니다.
sftp -P "$SFTP_PORT" -i "$PRIVATE_KEY_PATH" -o BatchMode=yes "$SFTP_USER@$SFTP_HOST" <<EOF
cd "$SFTP_REMOTE_DIR"
lcd "$LOCAL_TEMP_DIR"
put *
bye
EOF

echo "-> SFTP 전송 완료."

# ==========================================
# 5. 로컬 임시 파일 및 S3 데이터 정리 (선택)
# ==========================================
echo "[4/4] 임시 파일 정리 중..."

# AP 서버 디스크 공간 확보를 위해 다운로드했던 파일 제거
rm -rf "$LOCAL_TEMP_DIR"

# (옵션) 다음 수행을 위해 S3 버킷 내의 임시 데이터도 지우고 싶다면 아래 주석을 해제하세요.
# aws s3 rm "$S3_BUCKET" --recursive

echo "🎉 모든 프로세스가 성공적으로 완료되었습니다."
```

### 💡 스크립트 작성 및 실행 시 핵심 체크포인트

   1. SFTP 자동화 인증 (SSH Key)
   * 스크립트가 사람의 개입 없이(비대화형) 실행되려면 SFTP 비밀번호 입력창이 뜨면 안 됩니다.
      * 반드시 AP 서버의 SSH 공개키(id_rsa.pub)를 타겟 SFTP 서버의 ~/.ssh/authorized_keys에 등록한 뒤, 스크립트 내 PRIVATE_KEY_PATH에 개인키 경로를 지정해 주세요.
   2. GZIP 옵션 활용 (선택)
   * UNLOAD 쿼리 내부에 GZIP을 추가했습니다. 대용량 데이터일수록 압축을 해야 S3 ⇄ AP 서버 ⇄ SFTP 서버 구간의 네트워크 전송 속도가 수배 이상 빨라지고 대역폭을 아낄 수 있습니다. 만약 압축되지 않은 순수 CSV를 원하시면 GZIP 단어를 지우시면 됩니다.
   3. 디스크 공간 확인
   * Redshift 메모리(RAM)는 전혀 쓰지 않지만, S3에서 데이터를 내려받는 LOCAL_TEMP_DIR(위 스크립트에서는 /tmp)의 디스크 남은 용량이 데이터 크기보다 여유가 있는지 미리 확인해야 합니다.

## 2. 서버의 보안 정책 등으로 인해 sshpass 같은 외부 패키지를 추가로 설치할 수 없다면, 많은 리눅스 배포판(Ubuntu, CentOS 등)에 기본 내장되어 있거나 최소 설치 버전이 아닐 때 기본적으로 제공되는 expect 도구를 활용해야 합니다.
expect는 비밀번호 입력창(password:)과 같은 대화형 프롬프트를 모니터링하다가, 해당 문구가 나타나면 자동으로 비밀번호를 주입해 주는 스크립팅 언어입니다.
전체적인 흐름(Redshift UNLOAD ➔ S3 다운로드 ➔ SFTP 전송)은 동일하며, SFTP 전송 부분만 expect 문법으로 처리한 대안 코드입니다. [1] 

### 📋 expect 활용형 SFTP 자동화 전체 스크립트
```
#!/bin/bash
# 에러 발생 시 즉시 스크립트 중단set -e
# ==========================================
# 1. 환경 설정 (사용자 환경에 맞게 수정)
# ==========================================

# Redshift 설정
export PGPASSWORD="your_redshift_password"
REDSHIFT_HOST="://amazonaws.com"
REDSHIFT_PORT="5439"
DB_NAME="your_db"
USER_NAME="your_user"

# S3 및 AWS 설정 (Redshift가 접근할 IAM Role)
S3_BUCKET="s3://your-bucket-name/unload_data/"
IAM_ROLE="arn:aws:iam::123456789012:role/YourRedshiftRole"

# 로컬 임시 디렉토리 (스크립트 실행 서버 내부 디스크)
LOCAL_TEMP_DIR="/tmp/redshift_unload_$(date +%Y%m%d%H%M%S)"
mkdir -p "$LOCAL_TEMP_DIR"

# 원격 SFTP 서버 설정 (비밀번호 방식)
SFTP_HOST="://remote-server.com"
SFTP_PORT="22"
SFTP_USER="sftp_username"
SFTP_PASS="sftp_password_here"                # SFTP 접속 비밀번호 명시
SFTP_REMOTE_DIR="/path/to/remote/target/dir"   # 목적지 디렉토리

# ==========================================
# 2. Redshift UNLOAD 실행 (S3로 내보내기)
# ==========================================
echo "[1/4] Redshift에서 S3로 UNLOAD 시작..."

psql -h "$REDSHIFT_HOST" -p "$REDSHIFT_PORT" -d "$DB_NAME" -U "$USER_NAME" -c "
UNLOAD ('SELECT * FROM large_table WHERE created_at >= \'2026-01-01\'')
TO '$S3_BUCKET'
IAM_ROLE '$IAM_ROLE'
FORMAT AS CSV
GZIP
ALLOWOVERWRITE
MANIFEST;
"
echo "-> UNLOAD 완료."

# ==========================================
# 3. S3에서 로컬 임시 디렉토리로 다운로드
# ==========================================
echo "[2/4] S3에서 AP 서버 로컬 디렉토리로 다운로드 중..."

aws s3 cp "$S3_BUCKET" "$LOCAL_TEMP_DIR" --recursive

echo "-> 다운로드 완료. (위치: $LOCAL_TEMP_DIR)"

# ==========================================
# 4. expect를 통해 원격 서버로 SFTP 전송 (비밀번호 자동화)
# ==========================================

echo "[3/4] 원격 서버($SFTP_HOST)로 SFTP 전송 시작 (expect 활용)..."

# Bash 스크립트 내부에 인라인(Heredoc) 형태로 expect 코드를 삽입합니다.
expect <<EOF
# 응답 대기 시간 설정 (초 단위, 대용량 파일 전송 시 넉넉하게 1800초(30분) 설정)
set timeout 1800

# SFTP 접속 명령어 실행 (처음 접속 시 yes/no 묻는 프롬프트 방지를 위해 옵션 추가)
spawn sftp -P $SFTP_PORT -o StrictHostKeyChecking=no $SFTP_USER@$SFTP_HOST

# 상황별 프롬프트 매칭
expect {
    # 1. 비밀번호 입력 프롬프트가 나오면 비밀번호 송신 (\r은 엔터 키)
    "*password:" {
        send "$SFTP_PASS\r"
        exp_continue
    }
    # 2. SFTP 프롬프트가 열리면 차례대로 명령어 실행
    "sftp>" {
        send "cd $SFTP_REMOTE_DIR\r"
        expect "sftp>"
        send "lcd $LOCAL_TEMP_DIR\r"
        expect "sftp>"
        send "put *\r"
        expect "sftp>"
        send "bye\r"
    }
    # 3. 타임아웃이나 연결 거부 발생 시 종료
    timeout {
        exit 1
    }
    eof {
        exit 1
    }
}
EOF

echo "-> SFTP 전송 완료."
# ==========================================
# 5. 로컬 임시 파일 정리
# ==========================================
echo "[4/4] 임시 파일 정리 중..."

rm -rf "$LOCAL_TEMP_DIR"

echo "🎉 모든 프로세스가 성공적으로 완료되었습니다."
```

### 💡 expect 구현 시 알아두어야 할 주의사항

   1. set timeout 1800 (전송 시간 확보)
   * expect는 기본적으로 전송이나 응답이 10초 동안 없으면 작업이 실패했다고 간주하고 스크립트를 강제 종료(타겟 타임아웃)해 버립니다. UNLOAD한 파일의 크기가 수백 MB ~ 수 GB 이상으로 크다면 파일이 올라가는 동안 10초가 훌쩍 넘어가므로, 반드시 set timeout 값을 1800(30분) 이상으로 넉넉하게 지정해야 대용량 전송 중에 끊기지 않습니다.
   2. exp_continue 활용
   * SSH 접속 환경이나 서버 보안 규칙에 따라 비밀번호를 요구하기 전에 다른 문구가 먼저 출력될 수 있습니다. exp_continue를 넣어두면 비밀번호를 입력한 후에도 다음 단계(sftp>)로 부드럽게 넘어가도록 흐름을 유지해 줍니다.
   3. 만약 expect 조차 설치되어 있지 않은 완전 순정(Minimal) 상태라면?
   * 시스템 제약이 극심하여 expect 명령어 조차 사용할 수 없다면, 패스워드 자동화는 불가능하므로 앞서 안내해 드린 일반 FTP(21번 포트) 스크립트로 전환하거나, 인프라 담당자에게 요청하여 SSH Key(공개키) 인증을 1회 등록하는 것이 가장 확실하고 안전한 해결책입니다.
